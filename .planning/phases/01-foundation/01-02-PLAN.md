    **Part B: Lenis smooth scroll verification**

    4. In `frontend/src/components/providers/SmoothScroll.tsx`, add scroll-to-top on route change to prevent Lenis from retaining scroll position across navigations:

       ```typescript
       "use client";

       import { ReactLenis, useLenis } from "lenis/react";
       import { usePathname } from "next/navigation";
       import { useEffect, type ReactNode } from "react";

       function ScrollToTop() {
         const lenis = useLenis();
         const pathname = usePathname();

         useEffect(() => {
           if (lenis) {
             lenis.scrollTo(0, { immediate: true });
           }
         }, [pathname, lenis]);

         return null;
       }

       export default function SmoothScroll({ children }: { children: ReactNode }) {
         return (
           <ReactLenis
             root
             options={{
               lerp: 0.1,
               duration: 1.2,
               smoothWheel: true,
             }}
           >
             <ScrollToTop />
             {children}
           </ReactLenis>
         );
       }
       ```

       This ensures that when the user navigates between routes, Lenis scrolls to the top immediately, preventing the "stuck at scroll position" bug common with smooth scroll libraries and Next.js App Router.
